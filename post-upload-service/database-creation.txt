#For cassandra in cqlsh
CREATE KEYSPACE socialNetworkKeyspace WITH replication = {
    'class': 'SimpleStrategy',
    'replication_factor': 3
};

USE socialNetworkKeyspace;

CREATE TABLE snippets (
    id UUID PRIMARY KEY,
    favicon VARCHAR,
    title VARCHAR,
    text_preview VARCHAR,
	link VARCHAR
);

CREATE TABLE posts (
    id UUID PRIMARY KEY,
    user_id INT,
    text VARCHAR,
    color_preload list<VARCHAR>,
    created_at timestamp,
    images_amount INT,
    likes_counter INT,
    snippet_state INT,
    post_is_ready BOOLEAN
);

CREATE TABLE recommendations (
    id UUID PRIMARY KEY,
    user_id INT,
    posts_ids set<UUID>,
    new_posts_ids set<UUID>,
    recommend_taken_count INT
);
